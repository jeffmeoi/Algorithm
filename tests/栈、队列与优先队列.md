# 栈、队列与优先队列

## 分析

详见Cpp Standard Library.

```cpp
//栈：后进先出规则
#include<stack>
stack<int> s; 
s.push(新元素);		//入栈 
s.top();			//获取栈顶元素 
s.pop();			//出栈 
s.empty();			//判断是否为空 
s.size();			//获取栈的尺寸

//队列：先进先出原则
#include<queue>
queue<int> q;
q.push(新元素);		//入队 
q.pop();			//出队 
q.front();			//取队首元素
q.back();			//取队末元素 
q.empty();			//判断是否为空 
q.size();	 		//获取队列尺寸 

//优先队列： 可用于不断取得最值
#include<queue>
priority_queue<int> pq;//越小的整数优先级越低的优先队列
pq.push(新元素)			//入队 
pq.pop();				//出队
pq.top();				//取队首元素 
priority_queue<int, vector<int>, cmp> pq1;
//eg:
struct cmp{
	bool operator()(const int a, const int b) const {		//a的优先级比b小时返回true 
		return a % 10 > b % 10;
	}
};
priority_queue<int, vector<int>, greater<int> > pq2; 

//双端队列:
#include <deque>
deque<int> d {1, 4, 6, 4, 2, 7, 5, 6, 9};
d.push_back(新元素);		//尾插入
d.push_front(新元素);		//首插入
d.pop_back();			//尾元素出队
d.pop_front();			//首元素出队
d.insert(d.begin() + 5, 10);	//指定位置插入
d.erase(c1.begin()+5);		//指定位置除去
d.clear();			//清空队列
d.emplace(位置,新元素);	     //指定位置构造新元素
d.front();			//获取首元素
d.back();			//获取尾元素
d.emplace_front(新元素);	      //首部构造新元素
d.emplace_back(新元素);	      //尾部构造新元素
d.size();			//队列尺寸
```
